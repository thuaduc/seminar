% ---------------------------------------------------------------- slide 2 ---
% Every number in this talk is a SWE-bench resolve rate, so the vocabulary is paid
% for once, before the first measurement is shown.
%
% The slide also plants the talk's ending: the test filter is what gives SWE-bench a
% grading oracle, the issue filter is what gives it an intent, and those are exactly
% the two stages Act II attacks. That plant is SPOKEN, not shown -- see script.md.
% A slide that says "remember these two things" telegraphs the payoff and spends the
% audience's attention on a callback they cannot use yet.
\begin{frame}{What SWE-bench measures}
  \small
  \begin{columns}[T]
    \begin{column}{0.53\textwidth}
      \textbf{One instance}~\cite{jimenez2024swebench} = a real merged pull
      request that
      \begin{enumerate}
        \item closes a linked GitHub issue, \emph{and}
        \item modifies the test suite.
      \end{enumerate}
      \medskip

      \textbf{The agent sees} the repository \emph{before} the merge, plus the
      issue text. It returns a patch.
      \medskip

      \textbf{The patch is graded} by running tests:
      \begin{itemize}
        \item \emph{fail-to-pass} must flip,
        \item \emph{pass-to-pass} must not regress.
      \end{itemize}
      \medskip

      \oracle{Resolve rate} = both hold, on the first patch (pass@1).
    \end{column}
    \begin{column}{0.45\textwidth}
      \footnotesize
      % NB: inside a column, \textwidth IS the column width -- not the slide's.
      \setlength{\tabcolsep}{3pt}
      \renewcommand{\arraystretch}{0.88}
      \scriptsize
      \begin{tabular}{@{}llp{0.50\textwidth}@{}}
        \toprule
        \textbf{Split} & \textbf{Size} & \textbf{Kept only} \\
        \midrule
        Full     & 2{,}294 & every scraped PR that touches tests \\
        Lite     & 300     & single-file patches; some issues leak the fix \\
        Verified & 500     & human-screened: clear issues, fair tests \\
        Pro      & 1{,}865 & multi-file, long-horizon; held-out repos \\
        \bottomrule
      \end{tabular}
      \footnotesize
      \medskip

      \alert{Rates are not comparable across splits.} Verified is the most
      \emph{curated} split, not the hardest.
    \end{column}
  \end{columns}
\end{frame}

% Deliberately thin. Refinement's SUBSTANCE is Act II -- the axis slide sorts these same

\begin{frame}{Refinement: what to do with the signal}
  \small
  \textbf{Where it sits} --- input: whatever execution returned. Output: a
  \emph{decision} --- repair the patch, re-plan, or stop.
  \medskip

  \begin{itemize}
    \item \textbf{Repair from the trace}~\cite{jiang2023selfevolve} --- the traceback
      goes straight back into the prompt.
    \item \textbf{Explain your own code}~\cite{chen2024selfdebug} --- discrepancies
      surface. Up to $+12$\,\%, but only where tests exist.
    \item \textbf{Sample $N$, rerank with a trained critic}~\cite{openhands2025verified}
      --- OpenHands on Verified: 60.6\,\% $\to$ \textbf{66.4\,\%}.
  \end{itemize}
  \vfill
  \begin{center}
    Agents improve their own code.
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Interface and context move scores as much as the model}
  \small
  \begin{enumerate}

    \item \textbf{They all build a purpose-built action interface} --- none hands the

  \end{enumerate}

% --------------------------------------------------------------- slide 11 ---
% NB: the frame body must not begin with a brace group --- beamer would read it as
% a frame subtitle and typeset it as a blue bold heading.
%
% The title is slide 10's question, restated: this slide IS the answer, and saying so
% in the title spares the audience the work of noticing. The previous title ("Only one
% signal can surprise the model") asserted the punchline using a word the slide never
% defined -- "surprise" is information-theoretic here (a signal the model could not have
% produced itself), and an audience meeting it cold reads it as hand-waving. The word is
% still worth having, because it is the ONLY word under which the three-way split holds:
% a trained critic *can* contradict the generator (that is why reranking pays), it just
% cannot bring in information from outside. So the term is now introduced by apposition
% in the oracle block, where the plain phrasing defines it -- and the conclusion slide,
% which leans on "cannot surprise", stays anchored.
\begin{frame}{What can tell the model it is wrong? Three answers}
  \small
  \begin{columns}[T]
    \begin{column}{0.32\textwidth}
      \begin{block}{Authoritative oracle}
        \footnotesize
        Test runner, compiler, human. \oracle{Ground truth.}
        \medskip

        It can tell the model something it did not already believe --- it can
        \oracle{surprise} it.
      \end{block}
    \end{column}
    \begin{column}{0.32\textwidth}
      \onslide<2->{
      \begin{block}{Learned / generated proxy}
        \footnotesize
        Trained critic, generated tests, LLM judge.
        \medskip

        Carries real information --- but never more than it was fitted with. A
        generated test \emph{runs} for real; the model still wrote the answer key.
        \proxy{Ceiling: its own fidelity.}
      \end{block}}
    \end{column}
    \begin{column}{0.32\textwidth}
      \onslide<3->{
      \begin{block}{Same-distribution self-critique}
        \footnotesize
        The model judging itself.
        \medskip

        \selfc{No information from outside the model} --- the critic is drawn from
        the distribution that produced the error~\cite{huang2024selfcorrect}.
      \end{block}}
    \end{column}
  \end{columns}
  \vfill
  \onslide<4->{
  \begin{center}
    \small
    An agent that could tell its wrong code from its right code by inspection
    would have written the right code in the first place.\\[0.2em]
    {\footnotesize\itshape It can \emph{reorganise} what the model believes.
     It cannot \emph{add} to it.}
  \end{center}}
\end{frame}

\begin{frame}[noframenumbering]{Backup: context and cost}
  \small
  \begin{itemize}
    \item \textbf{Longer windows do not dissolve the bottleneck.} Retaining full
      \emph{dependency} context works best; padding with loosely related code
      \emph{degrades} correctness~\cite{repoexec2025}.
      \medskip
    \item \textbf{Context explosion and semantic drift.} Appending every
      observation buries the critical early information~\cite{contextastool2025}.
      \medskip
    \item \textbf{Cost is quadratic in turns.} The whole prefix is re-sent each
      turn, so tokens grow as $O(n^2)$~\cite{turncontrol2025}. A property of the
      protocol, not an empirical finding; prompt caching cuts the constant, not
      the growth rate.
      \medskip
    \item \textbf{Resolve rate decouples from compute}~\cite{sweeffi2025}.
      Agentless leads on resolve rate (48\,\% with Qwen3-32B) at 83.1 API calls
      and 209.4\,s per issue. The frontier is accuracy \emph{per dollar}.
      \medskip
    \item \textbf{And verification is the human bottleneck.} In an RCT,
      experienced developers were 19\,\% \emph{slower} with early-2025 AI tools ---
      while believing they had been 20\,\% faster~\cite{metr2025rct}.
  \end{itemize}
\end{frame}
